5   package rpg.ui;
6   
7   import java.awt.Color;
8   import java.awt.Graphics;
9   import java.awt.image.BufferedImage;
10  import java.io.IOException;
11  import java.net.URISyntaxException;
12  import java.net.URL;
13  import javax.imageio.ImageIO;
14  import javax.swing.JPanel;
15  import rpg.map.EnemyTile;
16  import rpg.map.Mmap;
17  import rpg.map.PathWay;
18  import rpg.map.Wall;
19  
20  /**
21   *
22   * @author erikv
23   */
24  public class GameMap extends JPanel {
25  
26      public Mmap map;
27      private BufferedImage door;
28      private BufferedImage wall;
29      private BufferedImage pathWay;
30      private BufferedImage player;
31      private BufferedImage enemy;
32      private int playerPosx;
33      private int playerPosy;
34  
35      
36      /**
37       * Constructor
38       * @param url gets path to txt file that makes the map
39       * @throws java.io.IOException
40       * @throws java.net.URISyntaxException
41       */
42      public GameMap(String url) throws IOException, URISyntaxException {
43          loadItems();
44          setOpaque(true);
45          setBackground(Color.BLACK);
46          map = new Mmap(url);
47          map.makeMap();
48          playerPosx = 40;
49          playerPosy = 40;
50      }
51      /**
52       * Loads images to generate new map
53       */
54      public void loadItems() {
55          try {
56              door = ImageIO.read((this.getClass().getClassLoader().getResourceAsStream("resources/door.png")));
57              wall = ImageIO.read((this.getClass().getClassLoader().getResourceAsStream("resources/wall.png")));
58              pathWay = ImageIO.read((this.getClass().getClassLoader().getResourceAsStream("resources/floor.png")));
59              player = ImageIO.read((this.getClass().getClassLoader().getResourceAsStream("resources/hero.png")));
60              enemy = ImageIO.read((this.getClass().getClassLoader().getResourceAsStream("resources/enemy.png")));
61          } catch (IOException ex) {
62              System.out.println(ex);
63          }
64      }
65  
66      @Override
67      public void paintComponent(Graphics g) {
68          super.paintComponent(g);
69          drawGame(g);
70          g.dispose();
71      }
72  /**
73       * Draws gamemap based on the tilemap
74       * @param g Graphics parameter
75       */
76      public void drawGame(Graphics g) {
77          int posX;
78          int posY = 0;
79          for (int i = 0; i < 20; i++) {
80              posX = 0;
81              for (int k = 0; k < 30; k++) {
82  
83                  if (map.tileMap[k][i] instanceof EnemyTile) {
84                      g.drawImage(pathWay, posX, posY, this);
85                      if(map.tileMap[k][i].hasEnemy()){
86                        g.drawImage(enemy, posX, posY,this);
87                      }
88                  }else if (map.tileMap[k][i] instanceof PathWay) {
89                       g.drawImage(pathWay, posX, posY,this);
90                  } 
91                  else if (map.tileMap[k][i] instanceof Wall) {
92                      g.drawImage(wall, posX, posY, this);
93                  } else {
94                      g.drawImage(door, posX, posY, this);
95                  }
96                  posX += 40;
97              }
98              posY += 40;
99          }
100         g.drawImage(player, playerPosx, playerPosy, this);
101     }
102     
103 /**
104      * Updates player position on map
105      * @param posY position in y axle
106      * @param posX position in x axle
107      */
108     public void updatePlayerPosition(int posY, int posX) {
109         playerPosx = posX;
110         playerPosy = posY;
111     }
112     /**
113      * checks if the tile is passable
114      * @param posY position in y axle
115      * @param posX position in x axle
116      * @return returns true if tile is passable
117      */
118     public boolean passableTileInPoss(int posX, int posY){
119         return map.tileMap[posY/40][posX/40].isPassable();
120     }
121         /**
122      * checks if the tile has enemy
123      * @param posY position in y axle
124      * @param posX position in x axle
125      * @return returns true if tile has enemy
126      */
127     public boolean enemyTile(int posX, int posY){
128         return map.tileMap[posY/40][posX/40].hasEnemy();
129     }
130         /**
131      * checks if the tile is door
132      * @param posY position in y axle
133      * @param posX position in x axle
134      * @return returns true if tile is door
135      */
136         public boolean isDoor(int posX, int posY){
137         return map.tileMap[posY/40][posX/40].isDoor();
138     }
139 }
