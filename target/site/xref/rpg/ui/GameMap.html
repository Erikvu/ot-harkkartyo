5   package rpg.ui;
6   
7   import java.awt.Color;
8   import java.awt.Graphics;
9   import java.awt.image.BufferedImage;
10  import java.io.IOException;
11  import java.net.URISyntaxException;
12  import java.net.URL;
13  import javax.imageio.ImageIO;
14  import javax.swing.JPanel;
15  import rpg.map.Map;
16  import rpg.map.PathWay;
17  import rpg.map.Tile;
18  import rpg.map.Wall;
19  
20  /**
21   *
22   * @author erikv
23   */
24  public class GameMap extends JPanel {
25  
26      private Map map;
27      private BufferedImage door;
28      private BufferedImage wall;
29      private BufferedImage pathWay;
30      private BufferedImage player;
31      private int playerPosx;
32      private int playerPosy;
33  
34      public GameMap(URL url) throws IOException, URISyntaxException {
35          loadItems();
36          setOpaque(true);
37          setBackground(Color.BLACK);
38          map = new Map(url);
39          map.makeMap();
40          playerPosx = 40;
41          playerPosy = 40;
42      }
43  
44      public void loadItems() {
45          try {
46              door = ImageIO.read((getClass().getClassLoader().getResource("resources/door.png")));
47              wall = ImageIO.read((getClass().getClassLoader().getResource("resources/wall.png")));
48              pathWay = ImageIO.read((getClass().getClassLoader().getResource("resources/floor.png")));
49              player = ImageIO.read((getClass().getClassLoader().getResource("resources/hero.png")));
50          } catch (IOException ex) {
51              System.out.println(ex);
52          }
53      }
54  
55      @Override
56      public void paintComponent(Graphics g) {
57          super.paintComponent(g);
58          drawGame(g);
59          g.dispose();
60      }
61  
62      public void drawGame(Graphics g) {
63          int posX;
64          int posY = 0;
65          for (int i = 0; i < 20; i++) {
66              posX = 0;
67              for (int k = 0; k < 30; k++) {
68  
69                  if (map.tileMap[k][i] instanceof PathWay) {
70                      g.drawImage(pathWay, posX, posY, this);
71                  } else if (map.tileMap[k][i] instanceof Wall) {
72                      g.drawImage(wall, posX, posY, this);
73                  } else {
74                      g.drawImage(door, posX, posY, this);
75                  }
76                  posX += 40;
77              }
78              posY += 40;
79          }
80          g.drawImage(player, playerPosx, playerPosy, this);
81      }
82  
83      public void updatePlayerPosition(int posY, int posX) {
84          playerPosx = posX;
85          playerPosy = posY;
86      }
87  }
