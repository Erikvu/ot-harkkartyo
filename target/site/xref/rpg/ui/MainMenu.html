5   package rpg.ui;
6   
7   import java.awt.Color;
8   import java.awt.Graphics;
9   import java.awt.image.BufferedImage;
10  import java.io.IOException;
11  import javax.imageio.ImageIO;
12  import javax.swing.JPanel;
13  
14  /**
15   *
16   * @author erikv
17   */
18  public class MainMenu extends JPanel {
19  
20      private BufferedImage newGame;
21      private BufferedImage continueButton;
22      private BufferedImage quit;
23      private BufferedImage arrow1;
24      private BufferedImage arrow;
25      public int[] arrowLocation = new int[]{305, 405, 505};
26      public int arrowPos;
27  
28      /**
29       * Constructor that loads images to generate menuscreen.
30       */
31      public MainMenu() {
32          arrowPos = 0;
33          try {
34              newGame = ImageIO.read((this.getClass().getClassLoader().getResourceAsStream("resources/newGame.png")));
35              continueButton = ImageIO.read((this.getClass().getClassLoader().getResourceAsStream("resources/Continue.png")));
36              quit = ImageIO.read((this.getClass().getClassLoader().getResourceAsStream("resources/Quit.png")));
37              arrow1 = ImageIO.read((this.getClass().getClassLoader().getResourceAsStream("resources/ArrowLeft.png")));
38              arrow = ImageIO.read((this.getClass().getClassLoader().getResourceAsStream("resources/ArrowRight.png")));
39          } catch (IOException ex) {
40              System.out.println(ex);
41          }
42          setOpaque(true);
43          setBackground(Color.BLACK);
44      }
45  
46      @Override
47      public void paintComponent(Graphics g) {
48          super.paintComponent(g);
49          drawMainMenu(g);
50          g.dispose();
51      }
52  
53      /**
54       * Drawns menu.
55       */
56      public void drawMainMenu(Graphics g) {
57          g.drawImage(continueButton, 500, 300, this);
58          g.drawImage(newGame, 500, 400, this);
59          g.drawImage(quit, 500, 500, this);
60          drawArrows(g);
61      }
62  
63      /**
64       * Drawns selection arrows.
65       */
66      private void drawArrows(Graphics g) {
67          g.drawImage(arrow, 450, arrowLocation[arrowPos], this);
68          g.drawImage(arrow1, 700, arrowLocation[arrowPos], this);
69      }
70  
71      /**
72       * updates arrowposition.
73       *
74       * @param pos new arrow position
75       */
76      public void setArrowPos(int pos) {
77          if (pos >= 0 && pos < 3) {
78              arrowPos = pos;
79          }
80      }
81  }
